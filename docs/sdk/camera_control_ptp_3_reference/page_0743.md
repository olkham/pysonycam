GeneralSettingResultFile
This file is generated by the Camera and sent to the User App.
It is used to communicate the results of the settings applied by the “SDIGeneralSettingFile” to the User
App.
Example of GeneralSettingResultFile:
<?xml version="1.0" encoding="UTF-8"?>
<CameraSettingConfirmation>
<FileFormatVersion version="1"/>
<Group>
```c
    <Item id="0x01001000" preset="1">
      <Param id="0x01001F01" isMatch="True"/>
      <Param id="0x01001F02" isMatch="True"/>
      <Param id="0x01001203" isMatch="True"/>
      <Param id="0x01001F04" isMatch="True"/>
      <Param id="0x01001F05" isMatch="True"/>
      <Param id="0x01001206" isMatch="True"/>
      <Param id="0x01001F07" isMatch="True"/>
      <Param id="0x01001208" isMatch="True"/>
```
</Item>
```c
    <Item id="0x01002000" >
      <Param id="0x01002F01" isMatch="True"/>
```
</Item>
</Group>
<Group>
```c
    <Item id="0x01003000">
      <Param id="0x01003201" isMatch="False"/>
```
</Item>
</Group>
<Group>
```c
    <Item id="0x01004000">
      <Param id="0x01004201" isMatch="True"/>
```
</Item>
</Group>
</CameraSettingConfirmation>
